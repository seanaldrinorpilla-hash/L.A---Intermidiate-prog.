full_name = input("Enter your full name: ").strip()


name_parts = full_name.split()

if len(name_parts) < 2:
    print("Please enter at least a first and last name.")
else:
   
    birth_year = input("Enter your birth year (e.g., 1998): ").strip()
    
    if not birth_year.isdigit() or len(birth_year) != 4:
        print("Please enter a valid 4-digit birth year.")
    else:
        first_name = name_parts[0].lower()
        last_name = name_parts[-1].lower()

       
        username = first_name[:3] + last_name[:3] + birth_year[-2:]

        
        print("Generated username:", username)
